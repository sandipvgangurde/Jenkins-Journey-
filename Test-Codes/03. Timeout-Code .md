```groovy

pipeline {
    agent any  // Run this pipeline on any available agent

    stages {

        /* -------------------------------------------------------
         * Stage 1: Build
         * -------------------------------------------------------
         * This stage represents the build process of the pipeline.
         * You can add actual build commands instead of echo.
         * ------------------------------------------------------- */
        stage("Build") {
            steps {
                echo "Building the image"
            }
        }

        /* -------------------------------------------------------
         * Stage 2: Test (Parallel Execution)
         * -------------------------------------------------------
         * This stage runs multiple test stages in parallel.
         * Each test stage has its own timeout using a script block.
         * If a test exceeds its timeout, it will be aborted.
         * ------------------------------------------------------- */
        stage("Test") {
            parallel {

                /* ---------------------------------------------------
                 * Unit Test 1
                 * ---------------------------------------------------
                 * Timeout is set to 1 minute.
                 * Sleep 45 seconds to simulate a test that passes.
                 * --------------------------------------------------- */
                stage("Unit Test 1") {
                    steps {
                        script {
                            timeout(time: 1, unit: 'MINUTES') {
                                echo "This test should not need more than 60 seconds"
                                sh 'sleep 45'
                                echo "Test completed within 60 seconds"
                            }
                        }
                    }
                    post {
                        aborted {
                            // This runs if timeout is exceeded
                            echo "Unit Test 1 timed out"
                        }
                        failure {
                            // Runs if test failed for any other reason
                            echo "Unit Test 1 failed"
                        }
                    }
                }

                /* ---------------------------------------------------
                 * Unit Test 2
                 * ---------------------------------------------------
                 * Timeout is set to 1 minute.
                 * Sleep 90 seconds to intentionally cause timeout.
                 * The post-aborted block captures timeout messages.
                 * --------------------------------------------------- */
                stage("Unit Test 2") {
                    steps {
                        script {
                            timeout(time: 1, unit: 'MINUTES') {
                                echo "This test should not need more than 90 seconds"
                                sh 'sleep 90'
                                echo "This line will not be executed due to timeout"
                            }
                        }
                    }
                    post {
                        aborted {
                            // This runs because this test intentionally exceeds timeout
                            echo "❌❌❌❌ Unit Test 2 due to timed out ❌❌❌" 
                        }
                        failure {
                            // Runs if test fails for a reason other than timeout
                            echo "Unit Test 2 failed"
                        }
                    }
                }

            } // end parallel
        }

        /* -------------------------------------------------------
         * Stage 3: Deployment
         * -------------------------------------------------------
         * This stage represents deployment to the staging server.
         * Add real deployment commands here.
         * ------------------------------------------------------- */
        stage("Deployment in Staging server") {
            steps {
                echo "Running deployment on the staging server"
            }
        }
    }

    /* -----------------------------------------------------------
     * Global Post Actions
     * -----------------------------------------------------------
     * These actions run after the entire pipeline completion.
     * success: Pipeline succeeded
     * failure: One or more stages failed or timed out
     * ----------------------------------------------------------- */
    post {
        success {
            echo "Pipeline completed successfully"
        }
        failure {
            echo "Pipeline failed. Check logs for details"
        }
    }
}
